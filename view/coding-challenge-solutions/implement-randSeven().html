Return a random array of 7 numbers between 1 & 7

Easy Solution Using Math.rand


function seven () {
   arr = [];
   arr.length = 7;
   for (let i=0; i'<'arr.length; i++) {
       arr[i] = Math.floor(Math.random() * 7) + 1;
   }
   return arr
 }
